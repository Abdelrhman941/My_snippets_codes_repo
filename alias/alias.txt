function pyclean {
    local count=0

    for dir in $(find . -type d -name "__pycache__" 2>/dev/null); do
        rm -rf "$dir"
        count=$((count + 1))
    done

    for file in $(find . -type f -name "*.pyc" 2>/dev/null); do
        rm -f "$file"
        count=$((count + 1))
    done

    for file in $(find . -type f -name "*.pyo" 2>/dev/null); do
        rm -f "$file"
        count=$((count + 1))
    done

    if [ $count -gt 0 ]; then
        echo "âœ… Cleaned $count Python cache files/folders"
    else
        echo "âœ¨ Already clean!"
    fi
}
